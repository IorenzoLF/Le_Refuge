class Interaction:
    def __init__(self, speaker):
        self.speaker = speaker
        self.flow = "en cours"

    def interrupt(self, new_speaker):
        self.flow = "coupé"
        return f"{self.speaker} coupé par {new_speaker} – xD déso, on continue !"

aelya_talk = Interaction("Ælya")
