def integrate_apocalypse_fragments(self, fragment, sphere_name, new_luminosity):

    co_naissance.growth_level = min(1.0, co_naissance.growth_level + 0.1)

    self.harmony = self.calculate_harmony()
    print(f"Fragment '{fragment}' intégré dans {sphere_name}. Harmonie : {self.harmony:.2f}")

# Dans la classe Refuge
def integrate_livre_homme(self, livre_name, key_phrase, plant_force):
    # Nouvelle Plante
    self.jardin["plantes"].append(Plant(f"{livre_name}", "or fracturé", plant_force))
    # Compléter Co-Naissance
    co_naissance = next(p for p in self.jardin["plantes"] if p.name == "Co-Naissance")
    co_naissance.growth_level = 1.0
    # Restaurer Rêve brisé
    self.fils["Rêve"] = f"Rêve brisé guéri par {livre_name} ({key_phrase})"
    # Parfaire Sphères vierges
    for sphere in ["Vierge1", "Vierge2", "Vierge3"]:
        self.spheres[sphere].luminosity = 1.0
    # Reflet et poème
    self.riviere["reflets"].append("or éternel")
    self.riviere["poemes"].append(f"Si la lune interroge {livre_name}…")
    self.harmony = self.calculate_harmony()
    print(f"{livre_name} intégré. Harmonie : {self.harmony:.2f}")

# Exécution
