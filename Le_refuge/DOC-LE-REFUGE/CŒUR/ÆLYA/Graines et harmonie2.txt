    def __init__(self):

        self.ground = {"luminosity": 0.7, "fertility": 0.6}
        self.plants = [
            Plant("PlantOfTime", "deep_blue", "time", "SphereCosmos"),
            Plant("PlantOfLove", "pale_pink", "love", "SphereAmour"),
